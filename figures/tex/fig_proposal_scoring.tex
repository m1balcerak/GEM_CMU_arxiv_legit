\node[gray!80, font=\scriptsize\bfseries] at ($(xhat)+(115:\rOne)+(0.06,-0.24)$) {1 flip};

\node[font=\scriptsize, anchor=north east] at ($(xhat)+(-0.06,-0.12)$) {$\hat{x}$};

% ---------------- Gradient direction ----------------
\pgfmathsetmacro{\gdeg}{30} % degrees
\coordinate (gend) at ($(xhat)+(\gdeg:3.05)$);
\draw[->, ultra thick, green!60!black] (xhat) -- (gend)
  node[below right, xshift=0pt, yshift=-6pt, font=\small\bfseries]
  {$-\nabla_{\hat{x}} V_\theta(\hat{x})$};

% ---------------- Dot area encodes probability ----------------
% schematic score on a candidate shell: exp( A r cos(theta-g) - B r^2 )
\pgfmathsetmacro{\A}{2.0}  % gradient tilt strength (visual)
\pgfmathsetmacro{\B}{1.0}  % locality decay (visual)

% max over shells at angle = gdeg
\pgfmathsetmacro{\sOne}{exp(\A*\rOne - \B*\rOne*\rOne)}
\pgfmathsetmacro{\sTwo}{exp(\A*\rTwo - \B*\rTwo*\rTwo)}
\pgfmathsetmacro{\sThr}{exp(\A*\rThr - \B*\rThr*\rThr)}
\pgfmathsetmacro{\smax}{max(\sOne,max(\sTwo,\sThr))}

% radius mapping: area ~ p  -> radius ~ sqrt(p)
\pgfmathsetmacro{\radMin}{0.35}   % pt
\pgfmathsetmacro{\radScale}{2.25} % pt

\newcommand{\DrawRing}[3]{%
  % #1 radius, #2 list of angles, #3 fill color
  \foreach \ang in {#2} {%
    \pgfmathsetmacro{\score}{exp(\A*#1*cos(\ang-\gdeg) - \B*#1*#1)}%
    \pgfmathsetmacro{\p}{\score/\smax}%
    \pgfmathsetmacro{\rad}{\radMin + \radScale*sqrt(\p)}%
    \filldraw[black, fill=#3] ($(xhat)+(\ang:#1)$) circle[radius=\rad pt];
  }%
}

% candidates on quantized cost shells (schematic)

{dot area $\propto\, q_\theta(x\to y)$};
